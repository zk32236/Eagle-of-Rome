def test_no_singleton_in_game_state():

def test_fields_are_private():
    """检查新增实体字段是否为私有"""
    from src.core.entities.contract import Contract
    from src.core.entities.entities import Province
    
    for cls in [Contract, Province]:
        for attr_name in dir(cls):
            if not attr_name.startswith('_') and not callable(getattr(cls, attr_name)):
                # 允许的公共属性列表
                allowed = ['contract_id', 'name', 'description']  # 根据实际情况扩展
                if attr_name not in allowed:
                    assert False, f"{cls.__name__}.{attr_name} 应为私有字段"
