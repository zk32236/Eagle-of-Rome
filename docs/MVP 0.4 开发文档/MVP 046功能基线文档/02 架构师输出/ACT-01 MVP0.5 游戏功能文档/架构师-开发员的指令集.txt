def process_land_income(self) -> None:
    """处理私地收入和公地租金"""
    private_rate = self._config.get("economic_rules.private_land_income_per_unit")
    public_rate = self._config.get("economic_rules.public_land_rent_per_unit")
    
    # 私地收入
    for member in self.get_living_members():
        income = member.land_private * private_rate
        member._wealth += income
        self.log_event(f"私地收入: {member.name} +{income}")
    
    # 公地租金
    for province in self._provinces.values():
        rent = province.land_public * public_rate
        self._treasury += rent
        self.log_event(f"公地租金: {province.name} +{rent}")

def process_contract_revenues(self) -> None:
    """结算所有执行中合同的利润"""
    for contract in self._contracts.values():
        if not contract.is_under_execution:
            continue
        
        # 根据合同类型计算利润
        if contract.contract_type == ContractType.TAX:
            profit = contract.profit_base  # 已在中标时计算
        else:  # PROJECT
            profit = contract.profit_base
        
        # 支付给中标骑士
        knight = self.get_member(contract.winner_id)
        if knight:
            knight.settle_contract_profit(profit)
            self.log_event(f"合同利润: {knight.name} 合同{contract.contract_id} +{profit}")

def _handle_tax_bid(self, contract, player, amount):
    """处理包税权竞标（价高者得）"""
    try:
        # 校验底价
        if amount < contract.base_cost:
            print(f"❌ 出价不得低于底价 {contract.base_cost}")
            return False
        
        # 校验财富
        if player.wealth < amount:
            print(f"❌ 个人财富不足，当前 {player.wealth}")
            return False
        
        # 执行资金转移（原子操作）
        player._wealth -= amount
        self.state._treasury += amount
        
        # 计算利润基数
        profit_rate = self.state._config.get("economic_rules.tax_contract_profit_rate")
        profit_base = int(contract.base_cost * profit_rate)
        
        # 更新合同状态
        contract.mark_winner(player.figure_id, self.state.turn, profit_base)
        
        # 更新人物关联
        player.add_contract(contract.contract_id)
        
        # 更新行省关联
        province = self.state.get_province(contract.province_id)
        province.bind_tax_contract(contract.contract_id)
        
        # 记录日志
        self.state.log_event(
            f"骑士 {player.name} 以 {amount} 塔兰特中标包税权合同 {contract.contract_id}"
        )
        print(f"✅ 竞标成功！")
        return True
        
    except Exception as e:
        # 回滚逻辑（实际可简化，因异常前操作少）
        print(f"❌ 竞标失败: {str(e)}")
        return False

def _handle_project_bid(self, contract, player, amount):
    """处理公共工程竞标（价低者得）"""
    try:
        # 校验上限
        if amount > contract.base_cost:
            print(f"❌ 出价不得高于招标底价 {contract.base_cost}")
            return False
        
        # 计算预付款
        advance_rate = self.state._config.get("economic_rules.project_advance_payment_rate")
        advance = int(contract.base_cost * advance_rate)
        
        # 校验国库资金
        if self.state._treasury < advance:
            print(f"❌ 国库资金不足支付预付款，需要 {advance}")
            return False
        
        # 执行资金转移
        self.state._treasury -= advance
        player._wealth += advance
        
        # 计算利润基数
        profit_rate = self.state._config.get("economic_rules.project_contract_profit_rate")
        profit_base = int(contract.base_cost * profit_rate)
        
        # 更新状态
        contract.mark_winner(player.figure_id, self.state.turn, profit_base)
        player.add_contract(contract.contract_id)
        
        province = self.state.get_province(contract.province_id)
        province.bind_project_contract(contract.contract_id)
        province._has_project = True
        
        self.state.log_event(
            f"骑士 {player.name} 中标工程合同 {contract.contract_id}，获预付款 {advance}"
        )
        print(f"✅ 竞标成功！获得预付款 {advance}")
        return True
        
    except Exception as e:
        print(f"❌ 竞标失败: {str(e)}")
        return False
