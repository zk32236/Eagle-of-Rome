# CommandRegistry自动注册伪代码

# 说明: 标注关键步骤注释，可直接指导编码

import os
import importlib.util
from pathlib import Path
from typing import Dict, List, Optional, Type

class CommandRegistry:
    """命令注册中心 - 自动扫描、注册、执行"""
    
    def __init__(self, commands_dir: str):
        self._commands: Dict[str, Type[Command]] = {}  # name -> CommandClass
        self._scan_and_register(commands_dir)  # 启动时自动扫描
    
    # ========== 核心步骤1: 扁平扫描 ==========
    def _scan_and_register(self, directory: str):
        """扫描目录下所有命令文件并注册"""
        # 关键注释: 使用Path.glob仅扫描当前目录，不递归子目录（扁平化要求）
        for file_path in Path(directory).glob("*.py"):
            # 关键注释: 跳过私有文件（下划线开头），如__init__.py, _base.py
            if file_path.name.startswith("_"):
                continue
            
            # 关键注释: 步骤2 - 动态导入，失败隔离（单文件失败不影响其他）
            module = self._load_module(file_path)
            if not module:
                continue  # 导入失败则跳过，继续扫描其他文件
            
            # 关键注释: 步骤3 - 提取Command子类（单文件单命令约定）
            for cmd_class in self._extract_commands(module):
                # 关键注释: 步骤4 - 注册并检测重复（冲突抛异常，阻止启动）
                self._register(cmd_class)
    
    # ========== 核心步骤2: 动态导入（规避循环导入） ==========
    def _load_module(self, file_path: Path) -> Optional[module]:
        """动态导入模块，异常隔离"""
        try:
            # 关键注释: 使用importlib.util动态导入，避免sys.modules污染
            module_name = file_path.stem  # 文件名作为模块名，如"phase_mortality"
            spec = importlib.util.spec_from_file_location(module_name, file_path)
            module = importlib.util.module_from_spec(spec)
            spec.loader.exec_module(module)  # 执行模块代码
            return module
        except Exception as e:
            # 关键注释: 打印警告但不阻断，确保其他命令正常加载
            print(f"[WARN] 命令文件导入失败 {file_path.name}: {e}")
            return None
    
    # ========== 核心步骤3: 类识别（过滤非命令类） ==========
    def _extract_commands(self, module) -> List[Type[Command]]:
        """从模块中提取Command子类"""
        commands = []
        for attr_name in dir(module):
            attr = getattr(module, attr_name)
            # 关键注释: 严格筛选条件：是类、继承Command、不是Command本身、有name属性
            if (isinstance(attr, type) and 
                issubclass(attr, Command) and 
                attr is not Command and
                hasattr(attr, 'name')):  # 验证必选属性存在
                commands.append(attr)
        return commands
    
    # ========== 核心步骤4: 注册与重复检测（Fail Fast） ==========
    def _register(self, cmd_class: Type[Command]):
        """注册命令，严格检测重复"""
        name = cmd_class.name
        
        # 关键注释: 主名冲突检测 - 已存在则抛异常，阻止启动（Fail Fast）
        if name in self._commands:
            existing_file = self._commands[name].__module__
            raise ValueError(f"命令名冲突: '{name}' 已存在于 {existing_file}")
        
        # 关键注释: 注册主名（存储类，非实例 - 延迟实例化）
        self._commands[name] = cmd_class
        
        # 关键注释: 别名冲突检测 - 与主名同等严格
        for alias in getattr(cmd_class, 'aliases', []):
            if alias in self._commands:
                existing_file = self._commands[alias].__module__
                raise ValueError(f"别名冲突: '{alias}' 已被 {existing_file} 使用")
            # 关键注释: 别名指向同一类，不创建副本
            self._commands[alias] = cmd_class
    
    # ========== 命令执行接口 ==========
    def execute(self, name: str, state: GameState, args: List[str]) -> bool:
        """执行指定命令"""
        cmd_class = self._commands.get(name)
        if not cmd_class:
            print(f"未知命令: {name}")
            return False
        
        # 关键注释: 延迟实例化 - 每次执行创建新实例，确保命令无状态残留
        instance = cmd_class(state)
        
        try:
            # 关键注释: 执行命令，异常内部消化（不向上抛）
            return instance.execute(args)
        except Exception as e:
            print(f"命令执行错误: {e}")
            return False
    
    def get_command_names(self) -> List[str]:
        """获取所有已注册命令名（去重，主名+别名）"""
        return list(self._commands.keys())

# ========== 关键设计决策摘要 ==========
# 1. 扁平扫描: Path.glob('*.py') 仅当前目录，无递归，符合约束
# 2. 动态导入: importlib.util 避免sys.modules污染，支持热重载
# 3. 单文件单命令: 通过hasattr(attr, 'name')过滤，非命令类自动忽略
# 4. 延迟实例化: 注册时存类，执行时new，保证命令实例无状态
# 5. 严格去重: name和alias冲突均抛ValueError，启动期发现，Fail Fast
# 6. 异常隔离: 单文件导入失败仅打印WARN，不影响其他命令加载
