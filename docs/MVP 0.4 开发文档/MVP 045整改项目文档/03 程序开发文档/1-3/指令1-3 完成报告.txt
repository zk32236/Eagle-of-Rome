6. src/core/game_state.py - 修复 get_living_members
python
# 原代码（占位实现）
def get_living_members(self) -> List['Figure']:
    return []

# 修改为（实际实现）
def get_living_members(self) -> List['Figure']:
    """获取所有存活人物"""
    return [m for m in self._members.values() if hasattr(m, 'is_dead') and not m.is_dead]

def get_member(self, member_id: int) -> Optional['Figure']:
    """获取人物"""
    return self._members.get(member_id)

8. src/ui/commands/func_load.py - 完善异常处理
python
def execute(self, args: List[str]) -> bool:
    try:
        ScenarioLoader.load_scenario(self.state, scenario_file)
        living_count = len(self.state.get_living_members())  # 获取实际存活人数
        print(f"\n✅ 场景加载成功: {scenario_file}")
        print(f"   国库: {self.state.treasury} 塔兰特")
        print(f"   存活人物: {living_count}")
        return True
    except FileNotFoundError as e:
        print(f"\n❌ 场景文件不存在: {scenario_file}")
        return False
    except Exception as e:
        print(f"\n❌ 场景加载失败: {e}")
        return False
